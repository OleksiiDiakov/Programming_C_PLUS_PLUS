#include <iostream>
using namespace std;

int main() {
    long long N;
    cout << "Enter your Number: ";
    cin >> N;

    int product = 1;
    bool hasOdd = false;

    while (N > 0) {
        int digit = N % 10;

        if (digit % 2) {
            product *= digit;
            hasOdd = true;
        }

        N/=10;
    }

    if (hasOdd)
        cout << "Product of odd numbers = " << product << endl;
    else
        cout << "There are no odd numbers." << endl;

    return 0;
}
