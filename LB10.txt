#include <iostream>
using namespace std;

int main() {
    const int m = 4; //ряд
    const int n = 3; //стовпець
    int a[m][n];

    cout << "Enter elements of 4x3 array:" << endl;
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            cin >> a[i][j]; //i ряд - j цифри у - ньому
        }
    }

    for (int i = 0; i < m; i++) {
        int count = 0;
        for (int j = 0; j < n; j++) {
            if (a[i][j] % 2 == 0) {
                count++;
            }
        }
        cout << "Row " << i << " has " << count << " even numbers." << endl;
    }

    return 0;
}

#include <iostream>
using namespace std;

int main() {
    const int m = 3;
    const int n = 4;
    int a[m][n];

    cout << "Enter elements of 3x4 array:" << endl;
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            cin >> a[i][j];
        }
    }

    int maxVal = a[0][0];
    int maxRow = 0; // m
    int maxCol = 0; //n

    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (a[i][j] > maxVal) {
                maxVal = a[i][j];
                maxRow = i;
                maxCol = j;
            }
        }
    }

    cout << "Max element: " << maxVal << endl;
    cout << "Position: row " << maxRow << ", column " << maxCol << endl;

    return 0;
}
