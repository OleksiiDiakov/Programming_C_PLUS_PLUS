#include <iostream>
using namespace std;

int main() {
    const int m = 12;
    int a[m] = {1, -11, 5, 2, -7, 7, 12, -2, 0, -6, 3, 4};

    for (int i = 0; i < m - 1; i++) {
        for (int j = 0; j < m - 1; j++) {
            if (a[j] > a[j + 1]) {
                int temp = a[j];
                a[j] = a[j + 1];
                a[j + 1] = temp;
            }
        }
    }

    cout << "Sorting in ascending order: ";
    for (int i = 0; i < m; i++) cout << a[i] << " ";
    cout << endl;

    for (int i = 0; i < m - 1; i++) {
        for (int j = 0; j < m - 1 - i; j++) {
            if (a[j] < a[j + 1]) {
                int temp = a[j];
                a[j] = a[j + 1];
                a[j + 1] = temp;
            }
        }
    }

    cout << "Sorting in descending order: ";
    for (int i = 0; i < m; i++) cout << a[i] << " ";
    cout << endl;

    return 0;
}

#include <iostream>
using namespace std;

int main() {
    const int m = 12;
    int a[m] = {1, -11, 5, 2, -7, 7, 12, -2, 0, -6, 3, 1};

    int max = a[0], maxIndex = 0;
    for (int i = 1; i < m; i++) {
        if (a[i] > max) {
            max = a[i];
            maxIndex = i;
        }
    }

    for (int i = 0; i < maxIndex; i++) {
        if (a[i] < 0) {
            a[i] = 0;
        }
    }

    cout << "Result: ";
    for (int i = 0; i < m; i++) cout << a[i] << " ";
    cout << endl;

    return 0;
}
