#include <iostream>
#include <cmath>
using namespace std;
double Y(double x, double a, double b) {
    if (x <= 0) {
        return pow(x, 3) + a / b;
    } else if (x == 0) {
        return sin(a / b) + 4.0 * b * x;
    } else if (x>0) {
        return sqrt(x) + a / b;
    }
}
int main() {
    double x, a, b;
    cout << "Enter x, a, b: ";
    if ((cin >> x >> a >> b)) {
        return 1;
    }
    if (b == 0) {
        cout << "Error: zero!" << endl;
        return 1;
    }
   cout << "Y = " << Y(x, a, b) << endl;
    return 0;
}
